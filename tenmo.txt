Tenmo app does the following:

list all users
path : /users
request method: GET
returns list of users 
succesful: 200
error: 400

list transfer history
path : /transferHistory
request method : GET
returns users transfer history
successful : 200
error : 400

get certain transfer history using id
path : /transferHistory/{id}
request method : GET
returns specific transfer history using id
successful : 200
error : 400

transfers money
path : /transfer
request method : POST
returns new balances for users
succesfull : 200
error : 400

see balance
path : /balance
request method : GET
returns users balance
successful : 200
error : 400

login
path : /login
request method: POST
returns logged in user
succesful : 200
error : 400

create user
path : /register
request method : CREATED
returns new user
succesfull : 200
error : 400
